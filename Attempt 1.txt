#include<bits/stdc++.h>

using namespace std;

typedef long long int ll;

int main()
{
    ll t;
    cin >> t;
    
    while(t--)
    {
        ll n,b;
        cin >> n >> b;
        
        ll a[n];
        
        for(ll i=0;i<n;i++)
            cin >> a[i];
            
        sort(a,a+n);
        
        ll ans =0;
        for(ll i=0;i<n;i++)
        {
            if(b>=a[i])
            {
                ans++;
                b-=a[i];
            }
            else
                break;
        }
        cout << ans << endl;
    }
    return 0;
}
